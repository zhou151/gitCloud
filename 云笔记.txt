ssh-keygen -t rsa -C 1045907858@qq.com  //生成密匙，在git bash里生成

git init //初始化,本地第一次使用

ssh -T git@github.com   //连接ssh      
-------------------------------------------------------------------------------------
         
git add . //添加所有文件到暂存区,未提交

git remote add origin git@github.com:zhou151/gitCloud.git   //添加仓库关联

git status  //查看提交状态

git commit -m '测试提交'  //提交暂存区的所有内容提交到当前分支上
-------------------------------------------------------------------------------------

git push -u origin master  //将本地项目推到远程仓库,-----仅限制为第一次上传项目----

-------------------------------------------------------------------------------------
git log //查看历史修改log

git remote -v  //仓库列表
-------------------------------------------------------------------------------------
git clone git@github.com:zhou151/gitCloud.git  //克隆云项目到本地

git merge origin/master  //合并远程+本地

git pull origin master //将本地项目推到远程仓库,不是第一次! git pull 也可以

git fetch origin master  //将远程修改更新到本地
-------------------------------------------------------------------------------------
