git status  //查看提交状态

git commit -m '测试提交'  //提交暂存区的所有内容提交到当前分支上

git push -u origin master  //将本地项目推到远程仓库

git pull --rebase origin master  //合并远程+本地

git remote -v  //仓库列表

git pull origin master //先拉下来，会自动合并的（不用操心）

git log //查看历史修改log
